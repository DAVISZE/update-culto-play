            protected void done() {
                splash.dispose();

    private void verificarConfiguracoes() {
        File arquivo = new File("welcome.dat");
        Properties prop = new Properties();
        if (arquivo.exists()) {
            try (FileInputStream in = new FileInputStream(arquivo)) {
                prop.load(in);
                corFundoApp = Color.decode(prop.getProperty("corFundo", "#141414"));
                corLetraProjetor = Color.decode(prop.getProperty("corLetra", "#FFFFFF"));
                tamanhoLetraProjetor = Integer.parseInt(prop.getProperty("tamanhoLetra", "60"));
                posicaoMargemProjetor = Integer.parseInt(prop.getProperty("margemY", "50"));
            } catch (Exception e) { aplicarPadroesIniciais(); }
        } else { abrirJanelaPersonalizacaoInicial(); }
    }

    private void salvarConfiguracoes() {

    public void transitarParaBiblia() {
        if (janelaBiblia.isVisible()) return;
        janelaBiblia.setOpacity(0f);
        janelaBiblia.setVisible(true);
        Timer timer = new Timer(15, e -> {
            float alfa = janelaBiblia.getOpacity() + 0.1f;
            if (alfa >= 1f) {
                janelaBiblia.setOpacity(1f);
                this.setVisible(false);
                ((Timer)e.getSource()).stop();
            } else { janelaBiblia.setOpacity(alfa); }
        });
        timer.start();
    }

    public void transitarParaConsole() {
        this.setVisible(true);
        Timer timer = new Timer(15, e -> {
            float alfa = janelaBiblia.getOpacity() - 0.1f;
            if (alfa <= 0f) {
                janelaBiblia.setVisible(false);
                janelaBiblia.setOpacity(1f);
                ((Timer)e.getSource()).stop();
            } else { janelaBiblia.setOpacity(alfa); }
        });
        timer.start();
    }

        campoBusca.setBackground(new Color(50, 50, 50));
        campoBusca.setForeground(Color.YELLOW);
        campoBusca.setCaretColor(Color.WHITE);
        campoBusca.setBorder(BorderFactory.createTitledBorder(null, "BUSCAR (TÍTULO OU LETRA)", 0, 0, null, Color.GRAY));

        modeloMusicas = new DefaultListModel<>();
        carregarListaArquivos("musicas", modeloMusicas);
        
        todasAsMusicas.clear();
        for(int i=0; i<modeloMusicas.size(); i++) todasAsMusicas.add(modeloMusicas.get(i));

        listaMusicas = new JList<>(modeloMusicas);
        configurarEstiloLista(listaMusicas);
        listaMusicas.addMouseListener(new MouseAdapter() {
            public void mouseClicked(MouseEvent e) {
                if(e.getClickCount() == 2) carregarMusica(listaMusicas.getSelectedValue());
            }
        });

        campoBusca.getDocument().addDocumentListener(new DocumentListener() {
            public void insertUpdate(DocumentEvent e) { filtrar(); }
            public void removeUpdate(DocumentEvent e) { filtrar(); }
            public void changedUpdate(DocumentEvent e) { filtrar(); }
            
            private void filtrar() {
                String busca = campoBusca.getText().toLowerCase();
                modeloMusicas.clear();
                for (String nome : todasAsMusicas) {
                    if (nome.toLowerCase().contains(busca)) {
                        modeloMusicas.addElement(nome);
                        continue;
                    }
                    try {
                        String conteudo = new String(Files.readAllBytes(new File("musicas/" + nome).toPath())).toLowerCase();
                        if (conteudo.contains(busca)) {
                            modeloMusicas.addElement(nome);
                        }
                    } catch (Exception ex) {}
                }
            }
        });

        painelMusicasCompleto.add(campoBusca, BorderLayout.NORTH);
        painelMusicasCompleto.add(new JScrollPane(listaMusicas), BorderLayout.CENTER);

    private void carregarMusica(String arq) {
        try {
            List<String> linhas = java.nio.file.Files.readAllLines(new File("musicas/" + arq).toPath());
            List<String> slides = new ArrayList<>();
            for (int i = 2; i < linhas.size(); i++) {
                if (!linhas.get(i).trim().isEmpty()) slides.add(linhas.get(i).trim());
            }
            if (modeloFundos != null && !modeloFundos.isEmpty()) {
                listaFundos.setSelectedIndex(0);
                projetor.setFundoGif("backgrounds/" + modeloFundos.get(0));
            }
            prepararMusica(slides);
            listaSlides.setBackground(new Color(30, 60, 30)); 
        } catch (Exception e) { e.printStackTrace(); }
    }

    public void prepararMusica(List<String> slides) {
        modeloSlides.clear();
        for (String s : slides) modeloSlides.addElement(s);
        if (!modeloSlides.isEmpty()) {
            listaSlides.setSelectedIndex(0);
            enviarTextoAoProjetor();
        }
    }

    private void enviarTextoAoProjetor() {
        String s = listaSlides.getSelectedValue();
        if (s != null) {
            pararLoopAnuncios();
            projetor.atualizarTexto(s);
            atualizarMonitor(s);
        }
    }

    private void executarBlackout() {
        pararLoopAnuncios();
        projetor.pararVideo(); 
        projetor.atualizarTexto("");
        projetor.limparFundo();
        atualizarMonitor("");
        listaSlides.setBackground(new Color(30, 30, 30));
    }

    private void projetarApresentacaoRapida() {
        String texto = areaTextoRapido.getText();
        String bg = (String) comboFundoRapido.getSelectedItem();
        if(bg.equals("Fundo Preto")) projetor.limparFundo();
        else projetor.setFundoGif("backgrounds/" + bg);
        projetor.atualizarTexto(texto);
        atualizarMonitor(texto);
    }

    private void exibirImagemDireta(String arq) {
        pararLoopAnuncios();
        projetor.pararVideo(); 
        projetor.limparFundo();
        projetor.setFundoGif("IMAGENS/" + arq);
        projetor.atualizarTexto("");
    }
    
    private void reproduzirVideo(String arq) {
        pararLoopAnuncios();
        projetor.atualizarTexto(""); 
        File videoFile = new File("VIDEOS/" + arq);
        String caminhoAbsoluto = videoFile.getAbsolutePath();
        if (!projetor.isVisible()) projetor.setVisible(true);
        projetor.tocarVideo(caminhoAbsoluto);
        atualizarMonitor("🎬 VÍDEO: " + arq);
    }

    private void carregarListaArquivos(String pasta, DefaultListModel<String> modelo) {
        File f = new File(pasta);
        if(!f.exists()) f.mkdir();

    }

    private JMenuBar criarMenuBar() {
        JMenuBar mb = new JMenuBar();
        JMenu m = new JMenu("ANÚNCIOS");
        JMenuItem i1 = new JMenuItem("▶️ Iniciar Loop");
        i1.addActionListener(e -> iniciarLoopAnuncios());
        JMenuItem i2 = new JMenuItem("⏹️ Parar");
        i2.addActionListener(e -> pararLoopAnuncios());
        m.add(i1); m.add(i2); mb.add(m);
        return mb;
    }

    private void iniciarLoopAnuncios() {
        pararLoopAnuncios();
        projetor.pararVideo();
        File pasta = new File("anuncios");
        if(!pasta.exists()) return;
        File[] arqs = pasta.listFiles();
        if(arqs == null || arqs.length == 0) return;
        listaArquivosAnuncios = Arrays.asList(arqs);
        timerAnuncios = new Timer(5000, e -> {
            if(indiceAnuncioAtual >= listaArquivosAnuncios.size()) indiceAnuncioAtual = 0;
            projetor.setFundoGif(listaArquivosAnuncios.get(indiceAnuncioAtual).getAbsolutePath());
            projetor.atualizarTexto("");
            indiceAnuncioAtual++;
        });
        timerAnuncios.start();
    }

    private void pararLoopAnuncios() {
        if(timerAnuncios != null) timerAnuncios.stop();
    }

    private void configurarAtalhosGlobais() {
        KeyboardFocusManager.getCurrentKeyboardFocusManager().addKeyEventDispatcher(e -> {
            if (e.getID() == KeyEvent.KEY_PRESSED) {
                int code = e.getKeyCode();
                if (code == KeyEvent.VK_F7) transitarParaBiblia();
                if (code == KeyEvent.VK_ESCAPE) executarBlackout();
                if (code == KeyEvent.VK_DOWN || code == KeyEvent.VK_RIGHT) {
                    int prox = listaSlides.getSelectedIndex() + 1;
                    if (prox < modeloSlides.size()) {
                        listaSlides.setSelectedIndex(prox);
                        listaSlides.ensureIndexIsVisible(prox);
                        enviarTextoAoProjetor();
                    }
                    return true; 
                }
                if (code == KeyEvent.VK_UP || code == KeyEvent.VK_LEFT) {
                    int prev = listaSlides.getSelectedIndex() - 1;
                    if (prev >= 0) {
                        listaSlides.setSelectedIndex(prev);
                        listaSlides.ensureIndexIsVisible(prev);
                        enviarTextoAoProjetor();
                    }
                    return true;
                }
            }
            return false;
        });
    }
